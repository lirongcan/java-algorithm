//图的边，增加两个设置函数
class edge implements Comparable<edge>{
    private int v;
    private double w;
    edge(int v,double w){this.v=v;this.w=w;}
    void setWeight(double w){this.w=w;}
    int vertex(){return v;}
    double weight(){return w;}
    @Override
    public int compareTo(edge x){
        return this.v-x.v;
    }
}
class graph{
    private int numv;
    private List<List<edge>>vertex;
    private void init(int n){
        numv=n;
        vertex=new ArrayList<>();
        for (int i = 0; i < n; i++) {
            vertex.add(new ArrayList<edge>());
        }
    }
    graph(int n){init(n);}
    void setEdge(int i,int j,int w){
        int temp=Collections.binarySearch(vertex.get(i),new edge(j,0));
        if(temp>=0)vertex.get(i).set(temp,new edge(j,w));
        else{
            temp=-temp-1;
            vertex.get(i).add(temp,new edge(j,w));
        }
    }
    double weight(int i,int j){
        int temp=Collections.binarySearch(vertex.get(i),new edge(j,0));
        if(temp>=0)return vertex.get(i).get(temp).weight();
        else return 0;
    }
    static class heap{
        private int maxsize,n;
        private edge[]h;
        //pos数组用来记录顶点v再堆数组h中的位置
        private int[]pos;
        //交换函数，要更新pos数组中的值
        private void swap(int i,int j){
            edge temp=h[i];
            h[i]=h[j];
            h[j]=temp;
            pos[h[i].vertex()]=i;
            pos[h[j].vertex()]=j;
        }
        //判断节点是否是叶子
        boolean isLeaf(int p){return (p>=n/2)&&p<n;}
        private void siftdown(int p){
            while(!isLeaf(p)){
                int l=2*p+1;
                int r=2*p+2;
                if(r<n&&h[r].weight()<h[l].weight())l=r;
                if(h[p].weight()<h[l].weight())return;
                swap(p,l);
                p=l;
            }
        }
        private void buildheap(){
            for (int i = h.length/2-1; i >= 0; i--)siftdown(i);
        }
        heap(edge[]hh,int n,int maxsize){
            h=hh;
            this.n=n;
            this.maxsize=maxsize;
            pos=new int[maxsize];
            for (int i = 0; i < h.length; i++) {
                pos[h[i].vertex()]=i;
            }
            buildheap();
        }
        void siftup(int i){
            while(i!=0&&h[i].weight()<h[(i-1)/2].weight()){
                swap(i,(i-1)/2);
                i=(i-1)/2;
            }
        }
        int remove(){
            swap(0,--n);
            if(n!=0)siftdown(0);
            return h[n].vertex();
        }
        //获取顶点v在堆数组中的位置
        int getPos(int v){
            return pos[v];
        }
        //获取堆数组位置为i的元素的权值，也就是与起点的距离
        double getWeight(int i){
            return h[i].weight();
        }
        //更新堆的函数，设置新的权值
        void setWeight(int i,double val){
            h[i].setWeight(val);
            //要调整堆的结构
            siftup(i);
            if(n!=0)siftdown(i);
        }
    }
    void dijkstra(){
        //创建用于堆的数组
        edge[]e=new edge[numv];
        //初始化这个数组，由于是对象数组所以要new一遍
        for (int i = 0; i < e.length; i++) {
            e[i]=new edge(i,Double.POSITIVE_INFINITY);
        }
        //把起点的内容添加进堆数组
        e[0].setWeight(0);
        heap p=new heap(e,e.length,e.length);
        //把邻接于起点的点加进堆中并更新与起点的距离（因为初始化的时候为无穷)
        for(int i=0;i<vertex.get(0).size();i++){
            p.setWeight(p.getPos(vertex.get(0).get(i).vertex()),weight(0,vertex.get(0).get(i).vertex()));
        }
        //第二个循环
        for (int i = 0; i < numv-1; i++) {
            //从堆中取出一个距离原点最近的点y
            int y=p.remove();
            //访问点y的邻点，如果点y到这个点的距离+点y到原点的距离<这个点到原点的距离
            //更新这个点到原点的距离
            for(int j=0;j<vertex.get(y).size();j++){
                if(p.getWeight(p.getPos(y))+vertex.get(y).get(j).weight()<p.getWeight(p.getPos(vertex.get(y).get(j).vertex()))){
                    p.setWeight(p.getPos(vertex.get(y).get(j).vertex()),p.getWeight(p.getPos(y))+vertex.get(y).get(j).weight());
                }
            }
        }
        //调试用，输出最后结果
        for (Algorithm.edge edge : e) {
            System.out.print(edge.vertex());
            System.out.print(",");
            System.out.println(edge.weight());
        }
    }
}

public class Kruskal {
    //用来实现kruskal算法的特殊的数据结构
    public static class graph{
        //存储边，这里的边包括两个端点和边的权值
        static class edge{
            int v1,v2,w;
            edge(int v1,int v2,int w){
                this.v1=v1;
                this.v2=v2;
                this.w=w;
            }
        }
        //使用并查集，算法取边是基于不生成环路的前提下
        private int[]rank;
        private int[]p;
        //用来每一次取一条边的小顶堆
        PriorityQueue<edge>heap;
        public graph(int n){
            heap=new PriorityQueue<>(new Comparator<edge>() {
                @Override
                public int compare(edge edge, edge t1) {
                    return edge.w-t1.w;
                }
            });
            rank=new int[n];
            p=new int[n];
            //初始化父亲数组
            for (int i = 0; i < n; i++) {
                p[i]=-1;
            }
        }
        //增加边
        public void add(int v1,int v2,int w){
            heap.add(new edge(v1,v2,w));
        }
        //从小顶堆中取边
        edge remove(){
            return heap.remove();
        }
        int find(int x){
            int temp=x;
            while(p[temp]!=-1)temp=p[temp];
            int root=temp;
            temp=x;
            while(p[temp]!=-1){
                int w=p[temp];
                p[temp]=root;
                temp=w;
            }
            return root;
        }
        boolean union(int x,int temp){
            int u=find(x),v=find(temp);
            if(u==v)return false;
            if(rank[u]<=rank[v]){
                p[u]=v;
                if(rank[u]==rank[v])rank[v]++;
            }
            else p[v]=u;
            return true;
        }
        public void kruskal(){
            int i=0;
            while(i<p.length-1){
                edge temp=remove();
                if(union(temp.v1,temp.v2)){
                    //调试用，输出最小生成树的边
                    System.out.print(temp.v1);
                    System.out.print(",");
                    System.out.print(temp.v2);
                    System.out.print(",");
                    System.out.println(temp.w);
                    i++;
                }
            }
        }
    }
}

public class Prim {
    //邻接表的边表示
    static class edge implements Comparable<edge>{
        private int v;
        private double w;
        edge(int v,double w){this.v=v;this.w=w;}
        @Override
        public int compareTo(edge edge) {
            return this.v-edge.v;
        }
    }
    //数据结构图
    static class graph{
        //顶点数量
        private int numv;
        //邻接表
        private List<List<edge>>vertex;
        private void init(int n){
            numv=n;
            vertex=new ArrayList<>();
            for (int i = 0; i < n; i++) {
                vertex.add(new ArrayList<edge>());
            }
        }
        graph(int n){init(n);}
        //设置或者添加边
        void setEdge(int i,int j,double w){
            int temp=Collections.binarySearch(vertex.get(i),new edge(j,0));
            if(temp>=0)vertex.get(i).set(temp,new edge(j,w));
            else{
                temp=-temp-1;
                vertex.get(i).add(temp,new edge(j,w));
            }
        }
        void prim(){
            //用neighbour数组表示X集中距离y最近的点
            int []neighbour=new int[numv];
            //用于堆的数组
            edge[]p=new edge[numv];
            //初始化全部为无穷
            for (int i = 0; i < numv; i++) {
                p[i]=new edge(i,Double.POSITIVE_INFINITY);
            }
            //更新与起点相邻节点的权值
            for (int i = 0; i < vertex.get(0).size(); i++) {
                    p[vertex.get(0).get(i).v].w=vertex.get(0).get(i).w;
            }
            //构建堆
            heap h=new heap(p,numv);
            for (int i = 0; i < numv-1; i++) {
                //从堆中取出一个顶点y
                int y=h.remove();
                System.out.print(neighbour[y]);
                System.out.print(",");
                System.out.print(y);
                System.out.print(",");
                System.out.println(h.getWeight(h.getPos(y)));
                //更新y的邻接点的权值
                for (int j = 0; j < vertex.get(y).size(); j++) {
                    if(vertex.get(y).get(j).w<h.getWeight(h.getPos(vertex.get(y).get(j).v))){
                        neighbour[vertex.get(y).get(j).v]=y;
                        h.setWeight(h.getPos(vertex.get(y).get(j).v),vertex.get(y).get(j).w);
                    }
                }
            }
        }
    }
    static class heap{
        private edge[]h;
        private int[]pos;
        //由于构建堆的时候是一个充满元素的堆，然后运行过程中只会让堆的元素越来越少
        //因此只需要用n来维护就行了，不需要maxsize
        private int n;
        private void swap(int i,int j){
            edge temp=h[i];
            h[i]=h[j];
            h[j]=temp;
            pos[h[i].v]=i;
            pos[h[j].v]=j;
        }
        boolean isLeaf(int p){return (p>=n/2)&&(p<n);}
        private void siftdown(int p){
            while(!isLeaf(p)){
                int l=2*p+1;
                int r=2*p+2;
                if(r<n&&h[l].w>h[r].w)l=r;
                if(h[p].w<h[l].w)return;
                swap(p,l);
                p=l;
            }
        }
        private void buildheap(){
            for (int i = h.length/2-1; i >= 0; i--) {
                siftdown(i);
            }
        }
        heap(edge[]hh,int n){
            this.n=n;
            h=hh;
            pos=new int[h.length];
            for (int i = 0; i < pos.length; i++) {
                pos[h[i].v]=i;
            }
            buildheap();
        }
        void siftup(int i){
            while(i!=0&&h[i].w<h[(i-1)/2].w){
                swap(i,(i-1)/2);
                i=(i-1)/2;
            }
        }
        int remove(){
            swap(0,--n);
            if(n!=0)siftdown(0);
            return h[n].v;
        }
        int getPos(int i){
            return pos[i];
        }
        double getWeight(int i){return h[i].w;}
        void setWeight(int i,double val){
            h[i].w=val;
            siftup(i);
            if(n!=0)siftdown(i);
        }
    }
    public void test(){
        graph t=new graph(6);
        t.setEdge(0,1,1);
        t.setEdge(0,2,12);
        t.setEdge(1,2,9);
        t.setEdge(1,3,3);
        t.setEdge(2,4,5);
        t.setEdge(3,2,4);
        t.setEdge(3,5,15);
        t.setEdge(3,4,13);
        t.setEdge(4,5,4);
        t.prim();
    }
}
